Calcule a média de cada variável e a média de Z, em que:

\[ z=\frac{w+i}{x} \]

R

# Calcular a média de cada variável
mean_w <- mean(df$w)
mean_x <- mean(df$x)
mean_i <- mean(df$i)

# Calcular Z
df$z <- (df$w + df$i) / df$x

# Calcular a média de Z
mean_z <- mean(df$z)

  mutate(Embarked = case_when(

  pivot_wider(
